/**
 * Distribution PM Suite — Google Apps Script Backend v2
 */
function corsOutput(data) {
  return ContentService.createTextOutput(JSON.stringify(data))
    .setMimeType(ContentService.MimeType.JSON);
}

function doGet(e) {
  try {
    const type = e.parameter.type;
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    let result = [];
    if (type === 'ping') return corsOutput({ ok: true });
    if (type === 'history') {
      const sheet = ss.getSheetByName('History');
      if (sheet && sheet.getLastRow() > 1) {
        const rows = sheet.getRange(2,1,sheet.getLastRow()-1,6).getValues();
        result = rows.filter(r=>r[0]).map(r=>({id:r[0],date:r[1],toolType:r[2],vendor:r[3],title:r[4],sub:r[5]}));
      }
    }
    if (type === 'deals') {
      const sheet = ss.getSheetByName('Deals');
      if (sheet && sheet.getLastRow() > 1) {
        const rows = sheet.getRange(2,1,sheet.getLastRow()-1,10).getValues();
        result = rows.filter(r=>r[0]).map(r=>({id:r[0],vendor:r[1],reseller:r[2],customer:r[3],product:r[4],amount:r[5],status:r[6],regDate:r[7],closeDate:r[8],note:r[9]}));
      }
    }
    if (type === 'mdf') {
      const sheet = ss.getSheetByName('MDF');
      if (sheet && sheet.getLastRow() > 1) {
        const rows = sheet.getRange(2,1,sheet.getLastRow()-1,9).getValues();
        result = rows.filter(r=>r[0]).map(r=>({id:r[0],vendor:r[1],category:r[2],title:r[3],amount:r[4],status:r[5],date:r[6],partner:r[7],note:r[8]}));
      }
    }
    if (type === 'mdf_budget') {
      const sheet = ss.getSheetByName('Config');
      if (sheet) {
        const data = sheet.getRange('B1:B4').getValues();
        return corsOutput({ok:true,data:{nvidia:Number(data[0][0])||0,dell:Number(data[1][0])||0,unity:Number(data[2][0])||0,other:Number(data[3][0])||0}});
      }
    }
    return corsOutput({ ok: true, data: result });
  } catch(err) { return corsOutput({ ok: false, error: err.message }); }
}

function doPost(e) {
  try {
    const data = JSON.parse(e.postData.contents);
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    if (data.type === 'history') {
      const sheet = ensureSheet(ss,'History',['ID','날짜','유형','벤더','제목','요약']);
      sheet.appendRow([data.id,data.date,data.toolType,data.vendor||'',data.title,data.sub||'']);
    }
    if (data.type === 'deals') {
      const sheet = ensureSheet(ss,'Deals',['ID','벤더','파트너사','고객사','제품/솔루션','예상금액(만원)','상태','등록일','예상종결일','메모']);
      sheet.clearContents();
      sheet.appendRow(['ID','벤더','파트너사','고객사','제품/솔루션','예상금액(만원)','상태','등록일','예상종결일','메모']);
      (data.deals||[]).forEach(d=>sheet.appendRow([d.id,d.vendor,d.reseller||'',d.customer||'',d.product||'',d.amount||'',d.status,d.regDate||'',d.closeDate||'',d.note||'']));
    }
    if (data.type === 'mdf') {
      const sheet = ensureSheet(ss,'MDF',['ID','벤더','카테고리','활동명','금액(만원)','상태','일정','파트너사','메모']);
      sheet.clearContents();
      sheet.appendRow(['ID','벤더','카테고리','활동명','금액(만원)','상태','일정','파트너사','메모']);
      (data.items||[]).forEach(i=>sheet.appendRow([i.id,i.vendor,i.category||'',i.title||'',i.amount||'',i.status,i.date||'',i.partner||'',i.note||'']));
    }
    if (data.type === 'mdf_budget') {
      const sheet = ensureSheet(ss,'Config',['항목','값']);
      sheet.getRange('A1:B4').setValues([['NVIDIA MDF 예산(만원)',data.nvidia||0],['Dell MDF 예산(만원)',data.dell||0],['Unity MDF 예산(만원)',data.unity||0],['기타 예산(만원)',data.other||0]]);
    }
    return corsOutput({ ok: true });
  } catch(err) { return corsOutput({ ok: false, error: err.message }); }
}

function ensureSheet(ss, name, headers) {
  let sheet = ss.getSheetByName(name);
  if (!sheet) {
    sheet = ss.insertSheet(name);
    sheet.appendRow(headers);
    sheet.getRange(1,1,1,headers.length).setFontWeight('bold').setBackground('#e8f0fe');
  }
  return sheet;
}
